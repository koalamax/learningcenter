(2) 테두리 사각형 코드


void main() {
  int size = 10;

  for (int i = 0; i < size; i++) {
    if (i == 0 || i == size - 1) {
      print('*' * size);
    } else {
      print('*' + ' ' * (size - 2) + '*');
    }
  }
}

(5) X표시 코드

void main() {
   int size = 10; 

  for (int i = 0; i < size; i++) {
    String line = '';
    for (int j = 0; j < size; j++) {
      if (j == i) {
        line += '*';
      } else if (j == size - i - 1) {
        line += '*';
      } else {
        line += ' ';
      }
    }
    print(line);
  }
}
